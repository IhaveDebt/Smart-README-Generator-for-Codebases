import os
import re
import json
from collections import defaultdict

class AutoDocPro:
    def __init__(self, root):
        self.root = os.path.abspath(root)
        self.languages = defaultdict(int)
        self.todos = []
        self.file_count = 0
        self.line_count = 0
        self.dependencies = set()

        self.ext_map = {
            ".py": "Python",
            ".js": "JavaScript",
            ".ts": "TypeScript",
            ".java": "Java",
            ".go": "Go",
            ".rs": "Rust",
            ".cpp": "C++",
            ".c": "C",
            ".cs": "C#",
            ".php": "PHP",
            ".rb": "Ruby"
        }

    def scan(self):
        for root, dirs, files in os.walk(self.root):
            for file in files:
                path = os.path.join(root, file)
                self.file_count += 1

                ext = os.path.splitext(file)[1]
                if ext in self.ext_map:
                    self.languages[self.ext_map[ext]] += 1

                try:
                    with open(path, "r", encoding="utf-8", errors="ignore") as f:
                        lines = f.readlines()
                        self.line_count += len(lines)

                        for i, line in enumerate(lines):
                            if "TODO" in line or "FIXME" in line:
                                self.todos.append({
                                    "file": path.replace(self.root, ""),
                                    "line": i + 1,
                                    "text": line.strip()
                                })

                            self.detect_dependencies(line)

                except:
                    pass

    def detect_dependencies(self, line):
        patterns = [
            r"import ([a-zA-Z0-9_]+)",
            r"from ([a-zA-Z0-9_]+) import",
            r"require\(['\"](.*?)['\"]\)"
        ]

        for p in patterns:
            m = re.search(p, line)
            if m:
                self.dependencies.add(m.group(1))

    def generate_tree(self):
        tree = []
        for root, dirs, files in os.walk(self.root):
            level = root.replace(self.root, "").count(os.sep)
            indent = "  " * level
            tree.append(f"{indent}{os.path.basename(root)}/")
            subindent = "  " * (level + 1)
            for f in files:
                tree.append(f"{subindent}{f}")
        return "\n".join(tree)

    def generate_language_section(self):
        out = []
        for lang, count in sorted(self.languages.items()):
            out.append(f"- {lang}: {count} files")
        return "\n".join(out) if out else "No known languages detected."

    def generate_todo_section(self):
        if not self.todos:
            return "No TODOs found."

        lines = []
        for t in self.todos[:20]:
            lines.append(f"- `{t['file']}:{t['line']}` — {t['text']}")
        return "\n".join(lines)

    def generate_dependency_section(self):
        if not self.dependencies:
            return "No imports detected."

        return "\n".join(f"- {d}" for d in sorted(self.dependencies))

    def generate_stats(self):
        return f"""
- Total Files: {self.file_count}
- Total Lines: {self.line_count}
- Languages Detected: {len(self.languages)}
"""

    def generate_readme(self):
        tree = self.generate_tree()
        langs = self.generate_language_section()
        todos = self.generate_todo_section()
        deps = self.generate_dependency_section()
        stats = self.generate_stats()

        return f"""
# AutoDoc Pro Documentation

## Overview
This documentation was automatically generated by **AutoDoc Pro**, a project analyzer
that scans the repository and extracts useful metadata.

---

## Project Statistics
{stats}

---

## Languages Used
{langs}

---

## Detected Dependencies
{deps}

---

## TODO / FIXME Notes
{todos}

---

## Project Structure
